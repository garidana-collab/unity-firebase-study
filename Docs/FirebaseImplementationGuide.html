FirebaseInitializer.cs 
private async UniTaskVoid InitializeFirebaseAsync()
{
    // 1. Firebase SDK 의존성 자동 수정
    DependencyStatus status =
        await FirebaseApp
            .CheckAndFixDependenciesAsync()
            .AsUniTask();

    if (status != DependencyStatus.Available)
    {
        Fail($"의존성 오류: {status}");
        return;
    }

    // 2. 핵심 인스턴스 확보
    App      = FirebaseApp.DefaultInstance;
    Database = GetDatabase(App); // FirebaseConfig에서 URL 읽음
    Auth     = FirebaseAuth.GetAuth(App);

    State = InitState.Ready;
}

// DB URL을 ScriptableObject에서 읽어 적용
private FirebaseDatabase GetDatabase(FirebaseApp app)
{
    FirebaseConfig config =
        Resources.Load<FirebaseConfig>("FirebaseConfig");

    if (config != null && !string.IsNullOrEmpty(config.databaseUrl))
        return FirebaseDatabase.GetInstance(app, config.databaseUrl);

    return FirebaseDatabase.GetInstance(app);
}

// 다른 매니저들이 대기할 때 사용하는 메서드
public async UniTask<bool> WaitForInitializationAsync()
{
    await UniTask.WaitUntil(() => State != InitState.Pending);
    return State == InitState.Ready;
}
🛠
FirebaseConfig (ScriptableObject): DB URL 등 설정값을 Resources/FirebaseConfig에 저장해 코드에서 하드코딩하지 않습니다. google-services.json은 Assets 루트에 배치합니다.

// 이메일 로그인 예시 (익명/회원가입도 동일한 try-catch 패턴)
public async UniTask<(bool success, string error)>
    SignInUserWithEmailAsync(string email, string pw)
{
    try
    {
        AuthResult result =
            await auth.SignInWithEmailAndPasswordAsync(email, pw);

        currentUser = result.User;
        NotifyLoginState(); // → LoginStatusChanged 이벤트 발행

        return (true, null);
    }
    catch (Exception ex)
    {
        // 영어 에러 원문 → 한국어로 파싱
        return (false, ParseFirebaseError(ex.Message));
    }
}

// Firebase 에러 메시지 → 한국어 변환
private string ParseFirebaseError(string error)
{
    string lower = error.ToLowerInvariant();

    if (lower.Contains("already in use"))   return "이미 사용 중인 이메일입니다.";
    if (lower.Contains("at least 6"))       return "비밀번호는 6자 이상이어야 합니다.";
    if (lower.Contains("badly formatted"))  return "이메일 형식이 올바르지 않습니다.";
    if (lower.Contains("network"))          return "네트워크 연결을 확인해주세요.";

    return "이메일 또는 비밀번호를 확인해주세요.";
}

ScoreManager.cs — SaveScoreAsync() 
private async UniTask<(bool success, string error)> SaveScoreAsync(int score)
{
    if (!AuthManager.Instance.IsLoggedIn)
        return (false, "로그인 필요");

    string userId = AuthManager.Instance.UserId;

    // ① 히스토리에 Push() — 자동 고유 키로 기록 추가
    DatabaseReference newHistoryRef =
        scoresRef.Child(userId).Child("history").Push();

    Dictionary<string, object> scoreData = new()
    {
        { "score",     score },
        { "timestamp", ServerValue.Timestamp }  // 서버 시간
    };
    await newHistoryRef.UpdateChildrenAsync(scoreData);

    // ② 신기록이면 bestscore 갱신 + 리더보드 반영
    if (score > cachedBestScore)
    {
        await UpdateBestScoreAsync(score);
        await LeaderboardManager.Instance.SaveToLeaderboard(score);
    }
}

실시간 리스너 (ValueChanged)

DB 데이터가 변경되는 순간 Firebase가 자동으로 콜백을 호출합니다. 다른 유저가 점수를 올리면 내 화면도 즉시 갱신됩니다.

리스너 시작/중지 
// 시작
public void StartRealtimeListener(int limit = 10)
{
    listenerQuery = leaderboardRef
        .OrderByChild("score")
        .LimitToLast(limit);

    listenerQuery.ValueChanged += OnValueChanged;
    isListenerActive = true;
}

// 중지 (씬 종료 / 패널 닫기)
public void StopRealtimeListener()
{
    listenerQuery.ValueChanged -= OnValueChanged;
    listenerQuery = null;
    isListenerActive = false;
}
⚠️
스레드 주의! Firebase의 ValueChanged 콜백은 백그라운드 스레드에서 호출됩니다. Unity UI는 메인 스레드에서만 접근 가능하므로 반드시 UniTask.SwitchToMainThread()가 필요합니다.
LeaderboardManager.cs — 실시간 콜백 처리 
// Firebase 백그라운드 스레드에서 호출됨
private void OnValueChanged(object sender, ValueChangedEventArgs args)
{
    if (args.DatabaseError != null)
    {
        Debug.Log($"[Leaderboard] 리스너 오류: {args.DatabaseError.Message}");
        return;
    }

    List<LeaderboardEntry> leaderboard = ParseEntries(args.Snapshot);
    DispatchUpdateAsync(leaderboard).Forget(); // 메인 스레드로 전환
}

private async UniTaskVoid DispatchUpdateAsync(List<LeaderboardEntry> leaderboard)
{
    await UniTask.SwitchToMainThread(); // ← Unity 메인 스레드로 복귀
    OnLeaderboardUpdated?.Invoke(leaderboard); // UI에 이벤트 전달
}
LeaderboardUI — UI 갱신 방식
1
패널 열릴 때 리스너 자동 시작 + 이벤트 구독
Start()에서 LeaderboardManager.Instance.StartRealtimeListener()를 호출하고 OnLeaderboardUpdated 이벤트를 구독합니다.
2
데이터 수신 시 기존 항목 전부 삭제 후 재생성
foreach (Transform child in leaderboardListParent) Destroy(child.gameObject)로 초기화 후 프리팹을 순위별로 Instantiate합니다.
3
Toggle로 실시간 동기화 ON/OFF
Toggle이 꺼지면 StopRealtimeListener(), 켜지면 StartRealtimeListener()를 호출합니다. 불필요한 DB 연결을 줄여 비용을 절감합니다.
4
OnDestroy에서 리스너 반드시 해제
LeaderboardManager.Instance.OnLeaderboardUpdated -= OnLeaderboardUpdated와 StopRealtimeListener()를 함께 호출해 메모리 누수를 방지합니다.

FirebaseValue.ToInt() 유틸: Firebase Realtime DB는 정수를 long 또는 double로 반환하는 경우가 있어, Unity의 int로 캐스팅할 때 타입 오류가 발생할 수 있습니다. 이를 방지하기 위해 FirebaseValue.ToInt(object value)에서 long, double, int 등 모든 경우를 처리합니다.
